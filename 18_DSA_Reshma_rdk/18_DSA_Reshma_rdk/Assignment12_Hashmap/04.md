
# Assignment — Student Database Using Hash Table (Student Generated)

## Title
Student Database Management System Using Hashing (Insert, Search, Delete)

## Objective
To write a program that simulates a student database using a hash table and supports:
- Insertion  
- Searching  
- Deletion  
of student records efficiently.

## Theory

A hash table stores data using a key-index mapping generated by a hash function.

### Key Used
**Roll Number**

### Hash Function
Using the divide method:
```
index = roll % size
```

### Collision Handling — Linear Probing
If collision occurs:
```
index = (index + 1) % size
```

### Student Record Structure
- Roll Number  
- Name  
- Course  

---

## Algorithm

### Insert
1. Compute index = roll % size  
2. If slot empty → insert  
3. Else → linear probe until empty slot found  

### Search
1. Compute index  
2. If not match → linear probe  
3. Stop if found or table fully scanned  

### Delete
1. Search record  
2. If found → mark as deleted  
3. Maintain deleted flag for probe continuity  

---

## Code (C++)
```cpp
#include <iostream>
#include <string>
using namespace std;

struct Student {
    int id;
    string name;
    string course;
    bool deleted; // flag to mark deletion
};

const int SIZE_rdk = 10; // hash table size
Student hashTable_rdk[SIZE_rdk];

// Hash function using MOD
int hashFunction_rdk(int key_rdk) {
    return key_rdk % SIZE_rdk;
}

// Insert student
void insertStudent_rdk() {
    Student s_rdk;
    cout << "Enter Student ID: ";
    cin >> s_rdk.id;
    cin.ignore();
    cout << "Enter Name: ";
    getline(cin, s_rdk.name);
    cout << "Enter Course: ";
    getline(cin, s_rdk.course);
    s_rdk.deleted = false;

    int index_rdk = hashFunction_rdk(s_rdk.id);
    int original_rdk = index_rdk;

    // Linear probing
    while (hashTable_rdk[index_rdk].id != 0 && !hashTable_rdk[index_rdk].deleted) {
        index_rdk = (index_rdk + 1) % SIZE_rdk;
        if (index_rdk == original_rdk) {
            cout << "Hash table is full!\n";
            return;
        }
    }

    hashTable_rdk[index_rdk] = s_rdk;
    cout << "Student inserted at index " << index_rdk << "\n";
}

// Search student
void searchStudent_rdk() {
    int id_rdk;
    cout << "Enter Student ID to search: ";
    cin >> id_rdk;
    int index_rdk = hashFunction_rdk(id_rdk);
    int original_rdk = index_rdk;

    while (hashTable_rdk[index_rdk].id != 0) {
        if (!hashTable_rdk[index_rdk].deleted && hashTable_rdk[index_rdk].id == id_rdk) {
            cout << "Student found at index " << index_rdk << ":\n";
            cout << "ID: " << hashTable_rdk[index_rdk].id << "\n";
            cout << "Name: " << hashTable_rdk[index_rdk].name << "\n";
            cout << "Course: " << hashTable_rdk[index_rdk].course << "\n";
            return;
        }
        index_rdk = (index_rdk + 1) % SIZE_rdk;
        if (index_rdk == original_rdk) break;
    }
    cout << "Student not found!\n";
}

// Delete student
void deleteStudent_rdk() {
    int id_rdk;
    cout << "Enter Student ID to delete: ";
    cin >> id_rdk;
    int index_rdk = hashFunction_rdk(id_rdk);
    int original_rdk = index_rdk;

    while (hashTable_rdk[index_rdk].id != 0) {
        if (!hashTable_rdk[index_rdk].deleted && hashTable_rdk[index_rdk].id == id_rdk) {
            hashTable_rdk[index_rdk].deleted = true;
            cout << "Student deleted from index " << index_rdk << "\n";
            return;
        }
        index_rdk = (index_rdk + 1) % SIZE_rdk;
        if (index_rdk == original_rdk) break;
    }
    cout << "Student not found!\n";
}

// Display table
void displayTable_rdk() {
    cout << "\nHash Table:\n";
    for (int i = 0; i < SIZE_rdk; i++) {
        cout << i << ": ";
        if (hashTable_rdk[i].id != 0 && !hashTable_rdk[i].deleted)
            cout << hashTable_rdk[i].id << " - " << hashTable_rdk[i].name << " - " << hashTable_rdk[i].course;
        cout << "\n";
    }
}

int main() {
    int choice_rdk;
    do {
        cout << "\n--- Student Database Menu ---\n";
        cout << "1. Insert Student\n";
        cout << "2. Search Student\n";
        cout << "3. Delete Student\n";
        cout << "4. Display Table\n";
        cout << "5. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice_rdk;

        switch(choice_rdk) {
            case 1: insertStudent_rdk(); break;
            case 2: searchStudent_rdk(); break;
            case 3: deleteStudent_rdk(); break;
            case 4: displayTable_rdk(); break;
            case 5: cout << "Exiting...\n"; break;
            default: cout << "Invalid choice!\n";
        }
    } while(choice_rdk != 5);

    return 0;
}

```
