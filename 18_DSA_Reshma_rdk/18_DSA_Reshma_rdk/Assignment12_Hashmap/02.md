
# Assignment — Faculty Database Simulation Using Hash Table (Student Generated)

## Title
Faculty Database Simulation Using Divide Hash Function, Linear Probing, and Chaining Without Replacement

## Objective
To write a program that simulates a faculty database using a hash table with:
- Divide (modulo) hash function  
- Linear probing  
- Chaining without replacement  
And to search faculty records based on faculty ID.

## Theory

A hash table stores elements using a key-index mapping.  
In this assignment:
- **Key** = Faculty ID  
- **Value** = Faculty Data (Name, Department)  
- **Hash Function** = Divide Method  
  ```
  index = FacultyID % size
  ```

### Collision Handling

### Linear Probing  
If the hashed index is occupied:
```
newIndex = (index + 1) % size
```

### Chaining Without Replacement  
- The original element at the hashed index is never moved.  
- Colliding records are inserted in the next available slot and linked using a `chain` index.

---

## Algorithm

### Insert Faculty
1. Compute index = ID % size  
2. If slot empty → insert  
3. Else find empty slot using linear probing  
4. Insert there and update chain  

### Search Faculty
1. Compute index  
2. If match → display  
3. Else follow chain[]  
4. If match found → display  
5. Else → not found

## Code (C++)
```cpp
#include <iostream>
#include <string>
using namespace std;

const int SIZE_rdk = 10;

struct Faculty_rdk {
    int id_rdk;
    string name_rdk;
    Faculty_rdk* next_rdk;   // for chaining
};

Faculty_rdk* hashTable_rdk[SIZE_rdk];

// ----------------------- INIT TABLE -----------------------
void initTable_rdk() {
    for (int i_rdk = 0; i_rdk < SIZE_rdk; i_rdk++) {
        hashTable_rdk[i_rdk] = NULL;
    }
}

// ----------------------- HASH FUNCTION -----------------------
int hashFunction_rdk(int key_rdk) {
    return key_rdk % SIZE_rdk;   // MOD method
}

// ----------------------- CREATE NODE -----------------------
Faculty_rdk* createNode_rdk(int id_rdk, string name_rdk) {
    Faculty_rdk* temp_rdk = new Faculty_rdk();
    temp_rdk->id_rdk = id_rdk;
    temp_rdk->name_rdk = name_rdk;
    temp_rdk->next_rdk = NULL;
    return temp_rdk;
}

// ----------------------- INSERT WITH REPLACEMENT -----------------------
void insertFaculty_rdk() {
    int id_rdk;
    string name_rdk;

    cout << "\nEnter Faculty ID: ";
    cin >> id_rdk;
    cout << "Enter Faculty Name: ";
    cin >> name_rdk;

    Faculty_rdk* newNode_rdk = createNode_rdk(id_rdk, name_rdk);

    int index_rdk = hashFunction_rdk(id_rdk);

    // Case 1 : Bucket empty
    if (hashTable_rdk[index_rdk] == NULL) {
        hashTable_rdk[index_rdk] = newNode_rdk;
        cout << "\nFaculty inserted successfully!";
        return;
    }

    // CASE 2 : CHAINING WITH REPLACEMENT

    int oldKey_rdk = hashTable_rdk[index_rdk]->id_rdk;
    int oldHome_rdk = hashFunction_rdk(oldKey_rdk);

    // Check if current element does NOT belong here
    if (oldHome_rdk != index_rdk) {
        // Swap and reinsert the old element
        Faculty_rdk* temp_rdk = hashTable_rdk[index_rdk];
        hashTable_rdk[index_rdk] = newNode_rdk;

        cout << "\nReplaced element at index " << index_rdk;

        // Reinsert the old element using linear probing
        newNode_rdk = temp_rdk;
        id_rdk = newNode_rdk->id_rdk;
        name_rdk = newNode_rdk->name_rdk;
    }

    // CASE 3 : LINEAR PROBING
    int start_rdk = index_rdk;

    do {
        index_rdk = (index_rdk + 1) % SIZE_rdk;

        if (hashTable_rdk[index_rdk] == NULL) {
            hashTable_rdk[index_rdk] = newNode_rdk;
            cout << "\nInserted using Linear Probing at index " << index_rdk;
            return;
        }

    } while (index_rdk != start_rdk);

    cout << "\nHash Table Full!";
}

// ----------------------- SEARCH -----------------------
void searchFaculty_rdk() {
    int id_rdk;
    cout << "\nEnter Faculty ID to Search: ";
    cin >> id_rdk;

    int index_rdk = hashFunction_rdk(id_rdk);
    int start_rdk = index_rdk;

    // Search entire table using probing + chaining
    do {
        Faculty_rdk* temp_rdk = hashTable_rdk[index_rdk];

        while (temp_rdk != NULL) {
            if (temp_rdk->id_rdk == id_rdk) {
                cout << "\nFaculty Found!";
                cout << "\nID: " << temp_rdk->id_rdk;
                cout << "\nName: " << temp_rdk->name_rdk << endl;
                return;
            }
            temp_rdk = temp_rdk->next_rdk;
        }

        index_rdk = (index_rdk + 1) % SIZE_rdk;

    } while (index_rdk != start_rdk);

    cout << "\nFaculty NOT found.\n";
}

// ----------------------- DISPLAY TABLE -----------------------
void displayTable_rdk() {
    cout << "\n\n---- FACULTY HASH TABLE ----\n";

    for (int i_rdk = 0; i_rdk < SIZE_rdk; i_rdk++) {
        cout << "[" << i_rdk << "] --> ";
        Faculty_rdk* temp_rdk = hashTable_rdk[i_rdk];

        if (temp_rdk == NULL) {
            cout << "EMPTY";
        }

        while (temp_rdk != NULL) {
            cout << "(ID: " << temp_rdk->id_rdk
                 << ", Name: " << temp_rdk->name_rdk << ") -> ";
            temp_rdk = temp_rdk->next_rdk;
        }

        cout << "NULL\n";
    }
}

// ----------------------- MAIN -----------------------
int main() {
    int choice_rdk;
    initTable_rdk();

    do {
        cout << "\n\n--- Faculty Database Menu ---\n";
        cout << "1. Insert Faculty\n";
        cout << "2. Search Faculty\n";
        cout << "3. Display Table\n";
        cout << "4. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice_rdk;

        switch (choice_rdk) {
            case 1: insertFaculty_rdk(); break;
            case 2: searchFaculty_rdk(); break;
            case 3: displayTable_rdk(); break;
            case 4: cout << "\nExiting...\n"; break;
            default: cout << "\nInvalid Choice!"; 
        }

    } while (choice_rdk != 4);

    return 0;
}

```
