#include <iostream>
#include <vector>
#include <climits>

using namespace std;

class Graph {
    int nodes_rdk;
    vector<vector<int>> adjMat_rdk;

public:
    Graph(int n_rdk) {
        nodes_rdk = n_rdk;
        adjMat_rdk.resize(nodes_rdk, vector<int>(nodes_rdk, 0));
    }

    void acceptGraph_rdk() {
        cout << "\nEnter Adjacency Matrix (0 for no edge):\n";
        for (int i = 0; i < nodes_rdk; i++) {
            for (int j = 0; j < nodes_rdk; j++) {
                cin >> adjMat_rdk[i][j];
            }
        }
        cout << "\nGraph accepted successfully!\n";
    }

    int minDistance_rdk(vector<int> &dist_rdk, vector<bool> &visited_rdk) {
        int min_rdk = INT_MAX, index_rdk = -1;

        for (int i = 0; i < nodes_rdk; i++) {
            if (!visited_rdk[i] && dist_rdk[i] < min_rdk) {
                min_rdk = dist_rdk[i];
                index_rdk = i;
            }
        }
        return index_rdk;
    }

    void dijkstra_rdk(int src_rdk, int dest_rdk) {
        vector<int> dist_rdk(nodes_rdk, INT_MAX);
        vector<bool> visited_rdk(nodes_rdk, false);

        dist_rdk[src_rdk] = 0;

        for (int i = 0; i < nodes_rdk - 1; i++) {
            int u_rdk = minDistance_rdk(dist_rdk, visited_rdk);
            visited_rdk[u_rdk] = true;

            for (int v_rdk = 0; v_rdk < nodes_rdk; v_rdk++) {

                if (!visited_rdk[v_rdk] &&
                    adjMat_rdk[u_rdk][v_rdk] != 0 &&
                    dist_rdk[u_rdk] != INT_MAX &&
                    dist_rdk[u_rdk] + adjMat_rdk[u_rdk][v_rdk] < dist_rdk[v_rdk]) {

                    dist_rdk[v_rdk] = dist_rdk[u_rdk] + adjMat_rdk[u_rdk][v_rdk];
                }
            }
        }

        cout << "\nShortest Distance from " << src_rdk 
             << " to " << dest_rdk << " = " << dist_rdk[dest_rdk] << "\n";
    }
};

int main() {
    int nodes_rdk, choice_rdk;
    int src_rdk, dest_rdk;

    cout << "Enter number of nodes: ";
    cin >> nodes_rdk;

    Graph g_rdk(nodes_rdk);

    while (true) {
        cout << "\n------ MENU ------\n";
        cout << "1. Accept Graph (Adjacency Matrix)\n";
        cout << "2. Find Shortest Path (Dijkstra)\n";
        cout << "3. Exit\n";
        cout << "Enter choice: ";
        cin >> choice_rdk;

        switch (choice_rdk) {
            case 1:
                g_rdk.acceptGraph_rdk();
                break;

            case 2:
                cout << "Enter source node: ";
                cin >> src_rdk;
                cout << "Enter destination node: ";
                cin >> dest_rdk;
                g_rdk.dijkstra_rdk(src_rdk, dest_rdk);
                break;

            case 3:
                cout << "Exiting...\n";
                return 0;

            default:
                cout << "Invalid choice! Try again.\n";
        }
    }
}
