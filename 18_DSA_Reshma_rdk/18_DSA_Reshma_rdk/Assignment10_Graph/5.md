#include <iostream>
#include <vector>
#include <queue>
#include <climits>
using namespace std;

struct Edge {
    int to, weight;
};

int main() {
    int n_rdk, e_rdk;
    cout << "Enter number of vertices: ";
    cin >> n_rdk;

    vector<vector<Edge>> adj_rdk(n_rdk);

    cout << "Enter number of edges: ";
    cin >> e_rdk;

    cout << "Enter edges (u v w):\n";
    for (int i = 0; i < e_rdk; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj_rdk[u].push_back({v, w});
        adj_rdk[v].push_back({u, w});     // remove if graph is directed
    }

    int src_rdk, dest_rdk;
    cout << "Enter source node: ";
    cin >> src_rdk;

    cout << "Enter destination node: ";
    cin >> dest_rdk;

    // Dijkstra
    vector<int> dist_rdk(n_rdk, INT_MAX);
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;

    dist_rdk[src_rdk] = 0;
    pq.push({0, src_rdk});

    while (!pq.empty()) {
        int d = pq.top().first;
        int node = pq.top().second;
        pq.pop();

        if (d > dist_rdk[node]) continue;

        for (auto &edge : adj_rdk[node]) {
            int next = edge.to;
            int w = edge.weight;

            if (dist_rdk[next] > dist_rdk[node] + w) {
                dist_rdk[next] = dist_rdk[node] + w;
                pq.push({dist_rdk[next], next});
            }
        }
    }

    cout << "\nShortest Distance from " << src_rdk 
         << " to " << dest_rdk << " = " << dist_rdk[dest_rdk] << endl;

    return 0;
}
